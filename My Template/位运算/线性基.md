# 线性基模板

#include<bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int inf = 1e18;

struct LinearBasis {
    vector<int> basis;
    
    LinearBasis() {
        basis.resize(60, 0);
    }
    
    void insert(int x) {
        for (int i = 59; i >= 0; i--) {
            if (!(x >> i & 1)) continue;
            if (!basis[i]) {
                basis[i] = x;
                return;
            }
            x ^= basis[i];
        }
    }
    
    bool canXOR(int k) {
        for (int i = 59; i >= 0; i--) {
            if (!(k >> i & 1)) continue;
            if (!basis[i]) return false;
            k ^= basis[i];
        }
        return true;
    }
    
    // 查询最大异或和
    int getMax() {
        int res = 0;
        for (int i = 59; i >= 0; i--) {
            if ((res ^ basis[i]) > res) {
                res ^= basis[i];
            }
        }
        return res;
    }
    
    // 查询最小异或和
    int getMin() {
        for (int i = 0; i <= 59; i++) {
            if (basis[i]) return basis[i];
        }
        return 0;
    }
    
    // 查询第 k 小异或和
    int kth(int k) {
        vector<int> v;
        for (int i = 0; i <= 59; i++) {
            if (basis[i]) {
                for (int j = i - 1; j >= 0; j--) {
                    if (basis[i] >> j & 1) {
                        basis[i] ^= basis[j];
                    }
                }
                v.push_back(basis[i]);
            }
        }
        
        if (v.size() == 0) return 0;
        if (k == 1) return 0;
        
        k--;
        if (k >= (1ll << v.size())) return -1;
        
        int res = 0;
        for (int i = 0; i < (int)v.size(); i++) {
            if (k >> i & 1) {
                res ^= v[i];
            }
        }
        return res;
    }
};

void solve() {
    int n, q;
    cin >> n >> q;
    
    LinearBasis lb;
    
    for (int i = 1; i <= n; i++) {
        int x;
        cin >> x;
        lb.insert(x);
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--)
    solve();
    return 0;
}
