#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;

int qpow(int a, int b) {
    int res = 1;
    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

void solve() {
    int n, m; cin >> n >> m;
    map<int, int> pw;
    for (int i = 0; i < n; i++) {
        int a, b; cin >> a >> b;
        pw[b] += a;
    }
    priority_queue<pair<int, int>> q;

    for (auto& [i, j] : pw) {
        if (j > m) {
            pw[i + 1] += (j - m + 1) / 2;
            if ((j - m) % 2) j = m - 1;
            else j = m;
        }
        if (j) q.push({i, j});
    }

    int res = 0, num = 0, exp = 0;
    while (!q.empty()) {
        auto [i, j] = q.top(); q.pop();
        if (num == 0) {
            num = m - j;
            exp = i;
            res = (res + qpow(2, exp)) % mod;
        } else {
            int dis = exp - i;
            exp = i;
            while (dis) {
                num <<= 1;
                dis--;
                if (num >= 2e9) break;
            }
            if (num >= 2e9) break;
            if (num < j) {
                num += m;
                res = (res + qpow(2, exp)) % mod;
            }
            num -= j;
        }
    }

    cout << res << '\n';
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) 
    solve();
}
