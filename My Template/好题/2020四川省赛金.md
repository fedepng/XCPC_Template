#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int INF = 1e18;
 
void solve() {
    int len = 1e9 + 1;
    int n, a, b; cin >> n >> a >> b;
    vector<int> p(n);
    vector<bool> d(n, 0);
    for (int& i : p) cin >> i;
    for (int i = 0; i < n; i++) {
        bool b; cin >> b;
        d[i] = b;
    }

    int ti = (min(a, b) - 1) / n;
    a -= ti * n;
    b -= ti * n;
    
    deque<int> q1, q2;
    for (int i = 0; i < n; i++) {
        if (d[i]) q2.push_front(len - p[i]);
        else q1.push_back(p[i]);
    }

    int res1 = ti * len * 2, res2 = 0;
    while (!q1.empty() || !q2.empty()) {
        if (!q1.empty()) {
            int p1 = q1.front();
            q1.pop_front();
            if (a > 0) q2.push_back(p1 + len);
            a--;
            res2 = max(res2, p1);
        }

        if (!q2.empty()) {
            int p2 = q2.front();
            q2.pop_front();
            if (b > 0) q1.push_back(p2 + len);
            b--;
            res2 = max(res2, p2);
        }
    }

    cout << res1 + res2 << '\n';
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    //int t; cin >> t;

    solve();
}
