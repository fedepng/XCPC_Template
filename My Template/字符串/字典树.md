# 字典树模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int inf = 1e18;

struct Trie {
    vector<array<int, 26>> nxt;
    vector<int> cnt;
    int idx;

    Trie() {
        nxt.emplace_back();
        cnt.push_back(0);
        idx = 0;
    }

    void insert(string& s) {
        int p = 0;
        for (char ch : s) {
            int c = ch - 'a';
            if (nxt[p][c] == 0) {
                nxt[p][c] = ++idx;
                nxt.emplace_back();
                cnt.push_back(0);
            }
            p = nxt[p][c];
        }
        cnt[p]++;
    }

    int query(string& s) {
        int p = 0;
        for (char ch : s) {
            int c = ch - 'a';
            if (nxt[p][c] == 0) return 0;
            p = nxt[p][c];
        }
        return cnt[p];
    }
};

void solve() {
    Trie trie;
    int n;
    cin >> n;
    
    for (int i = 0; i < n; i++) {
        string op, s;
        cin >> op >> s;
        
        if (op == "insert") {
            trie.insert(s);
        } else if (op == "query") {
            cout << trie.query(s) << '\n';
        }
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    //int t; cin >> t;
    //while (t--)
    solve();
    return 0;
}
```
