# 单调栈模板
可以以 $O(n)$ 复杂度求每个元素某侧第一个比它大/小的元素的下标或值。
```cpp
stack<int> st;
vector<int> ne_l, ne_r;
vector<int> a;

signed main() {
    cin >> n;
    ne_l.resize(n), ne_r.resize(n), a.resize(n);
    for (int& i : a) cin >> i;

    for (int i = 0; i < n; i++) {
        while (!st.empty() && a[i] >= a[st.back()]) {
            ne_r[st.back()] = i;
            st.pop();
        }
        st.push(i);
    }
    while (!st.empty()) {
        ne_r[st.back()] = n;
        st.pop();
    }

    for (int i = n - 1; i >= 0; i--) {
        while (!st.empty() && a[i] >= a[st.back()]) {
            ne_l[st.back()] = i;
            st.pop();
        }
        st.push(i);
    }
    while (!st.empty()) {
        ne_l[st.back()] = -1;
        st.pop();
    }
}
```
