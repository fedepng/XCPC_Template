# 组合数模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int inf = 1e18;
const int TABLE_SIZE = 1e6 + 10;

vector<int> fac(TABLE_SIZE);
vector<int> finv(TABLE_SIZE);

int exgcd(int a, int b, int &x, int &y) {
    if (b == 0) {
        x = 1, y = 0;
        return a;
    }
    int x1, y1;
    int g = exgcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - (a / b) * y1;
    return g;
}

int inv(int a) {
    int x, y;
    exgcd(a, mod, x, y);
    x %= mod;
    if (x < 0) x += mod;
    return x;
}

void init() {
    fac[0] = 1;
    for (int i = 1; i < TABLE_SIZE; i++) {
        fac[i] = fac[i - 1] * i % mod;
    }
    
    finv[TABLE_SIZE - 1] = inv(fac[TABLE_SIZE - 1]);
    for (int i = TABLE_SIZE - 2; i >= 0; i--) {
        finv[i] = finv[i + 1] * (i + 1) % mod;
    }
}

int C(int n, int k) {
    if (n < 0 || k < 0 || k > n) return 0;
    return fac[n] * finv[k] % mod * finv[n - k] % mod;
}

void solve() {
    int n, k;
    cin >> n >> k;
    cout << C(n, k) << '\n';
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    init();
    int t; cin >> t;
    while (t--)
    solve();
    return 0;
}
```
