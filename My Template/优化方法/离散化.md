vector<int> discretize(vector<int>& a, map<int, int>& mp) {
    vector<int> temp(a);
    sort(temp.begin(), temp.end());
    temp.erase(unique(temp.begin(), temp.end()), temp.end());

    vector<int> b;
    for (int i = 0; i < a .size(); i++) {
        b.push_back(lower_bound(temp.begin(), temp.end(), a[i]) - temp.begin());
        mp[a[i]] = b[i];
    }

    return b;
}
