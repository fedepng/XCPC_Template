# 欧拉筛求质数模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int inf = 1e18;
const int N = 1e7 + 10;

vector<int> prime;
bool vis[N];

void euler_sieve(int n) {
    for (int i = 2; i <= n; i++) {
        if (!vis[i]) prime.push_back(i);
        for (int j : prime) {
            if (i * j > n) break;
            vis[i * j] = 1;
            if (i % j == 0) break;
        }
    }
}

void solve() {
    int n; cin >> n;
    euler_sieve(n);
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    //int t; cin >> t;
    //while (t--)
    solve();
    return 0;
}
```
