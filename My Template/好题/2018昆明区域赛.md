#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;

int n, m;
vector<int> a;

bool check(int mid) {
    int sum = 0;
    vector<int> b(n + 1, 0);
    for (int i = 0; i < n; i++) {
        if (i == n - 1 && b[i] >= mid) break;
        b[i] += a[i];
        sum++;
        if (b[i] >= mid) continue;
        int cnt = (mid - b[i] + a[i] - 1) / a[i];
        sum += cnt * 2;
        b[i + 1] += cnt * a[i + 1];
        if (sum > m) return 0;
    }
    if (sum > m) return 0;
    else return 1;
}

void solve() {
    cin >> n >> m;
    a.clear();
    a.resize(n);
    for (int& i : a) cin >> i;

    int l = 0, r = 1e18;
    int mid, res = 0;
    while (l <= r) {
        mid = l + (r - l) / 2;
        if (check(mid)) {
            res = max(res, mid);
            l = mid + 1;
        } else r = mid - 1;
    }

    cout << res << '\n';
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) 
    solve();
}
