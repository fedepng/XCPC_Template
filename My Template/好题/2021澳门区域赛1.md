#include<bits/stdc++.h>
using namespace std;
#define int long long

struct Point {
    int x, y;
    
    int cross(const Point& b) const {
        return x * b.y - y * b.x;
    }
    
    int quad() const {
        if (x > 0 && y >= 0) return 1;
        if (x <= 0 && y > 0) return 2;
        if (x < 0 && y <= 0) return 3;
        return 4;  // x>=0 && y<0
    }
};

struct Node {
    int qd;
    Point v;
    
    bool operator<(const Node& b) const {
        if (qd != b.qd) return qd < b.qd;
        return v.cross(b.v) > 0;  // 极角从小到大，叉积为正则夹角小
    }
};

void solve() {
    int n;
    cin >> n;
    vector<Node> vec;
    
    for (int i = 0; i < n; i++) {
        Point p;
        cin >> p.x >> p.y;
        int q = p.quad();
        vec.push_back({q, p});
        vec.push_back({q + 4, p});  // 复制一份，便于处理循环
    }
    
    sort(vec.begin(), vec.end());
    
    int ans = n;
    
    for (int i = 0; i < n; i++) {
        Point tmp = vec[i].v;
        Node target = {vec[i].qd + 2, {-tmp.x, -tmp.y}};
        
        // 找到第一个 >= target 的位置
        int pos = lower_bound(vec.begin(), vec.end(), target) - vec.begin();
        int cnt = pos - i;  // 在 i 到 pos-1 范围内的点数（半平面内）
        
        ans = min({ans, cnt, n - cnt});
    }
    
    cout << ans << '\n';
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    
    return 0;
}
