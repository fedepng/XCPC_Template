# 离散化模板

```cpp
#include<bits/stdc++.h>
#define int long long
#define maxn 200005

const int INF = 1e18;

using namespace std;

int n;
vector<int> a, b, c, d;
map<int, int> Map;

void discret() {
    sort(d.begin(), d.end());
    for (int i = 0; i < d.size(); i++) {
        if (i == 0 || d[i] != d[i - 1]) {
            Map[d[i]] = c.size();
            c.push_back(d[i]);
        }
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    cin >> n;
    for (int i = 1; i <= n; i++) {
        int x, y;
        cin >> x >> y;
        a.push_back(x), b.push_back(y);
        d.push_back(x), d.push_back(y);
    }
}
```
