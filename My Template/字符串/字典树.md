#include<bits/stdc++.h>
#define int long long
#define maxn 200005

using namespace std;

const int INF = 1e18;

int trie[maxn][26];
int word[maxn];
int n, tot;

void insert(string s) {
    int u = 0;
    for (int i = 0; i < s.size(); i++) {
        int a = s[i] - 'a';
        if (trie[u][a] == 0) trie[u][a] = ++tot;
        u = trie[u][a];
    }
    word[u]++;
}
