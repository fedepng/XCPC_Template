# 线段树模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int INF = 1e18;
const int mod = 998244353;

struct SegmentTree {
    int n;
    vector<int> tree, lz;
    
    SegmentTree(int _n) : n(_n) {
        tree.resize(n << 2);
        lz.resize(n << 2);
    }
    
    void pushup(int rt) {
        tree[rt] = tree[rt << 1] + tree[rt << 1 | 1];
    }
    
    void pushdown(int rt, int l, int r) {
        if (lz[rt] == 0 || l == r) return;
        int mid = (l + r) >> 1;
        int left = rt << 1, right = rt << 1 | 1;
        
        tree[left] += lz[rt] * (mid - l + 1);
        lz[left] += lz[rt];
        tree[right] += lz[rt] * (r - mid);
        lz[right] += lz[rt];
        lz[rt] = 0;
    }
    
    void build(int rt, int l, int r, vector<int>& arr) {
        lz[rt] = 0;
        if (l == r) {
            tree[rt] = arr[l];
            return;
        }
        int mid = (l + r) >> 1;
        build(rt << 1, l, mid, arr);
        build(rt << 1 | 1, mid + 1, r, arr);
        pushup(rt);
    }
    
    void update(int rt, int l, int r, int L, int R, int val) {
        if (L <= l && r <= R) {
            tree[rt] += val * (r - l + 1);
            lz[rt] += val;
            return;
        }
        if (R < l || L > r) return;
        pushdown(rt, l, r);
        int mid = (l + r) >> 1;
        update(rt << 1, l, mid, L, R, val);
        update(rt << 1 | 1, mid + 1, r, L, R, val);
        pushup(rt);
    }
    
    int query(int rt, int l, int r, int L, int R) {
        if (L <= l && r <= R) return tree[rt];
        if (R < l || L > r) return 0;
        pushdown(rt, l, r);
        int mid = (l + r) >> 1;
        return query(rt << 1, l, mid, L, R) + query(rt << 1 | 1, mid + 1, r, L, R);
    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    
    vector<int> arr(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> arr[i];
    }
    
    SegmentTree seg(n);
    seg.build(1, 1, n, arr);
    
    while (m--) {
        int op, l, r, val;
        cin >> op;
        if (op == 1) {
            cin >> l >> r >> val;
            seg.update(1, 1, n, l, r, val);
        } else {
            cin >> l >> r;
            cout << seg.query(1, 1, n, l, r) << '\n';
        }
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) solve();
    return 0;
}
```
