# 匈牙利算法实现二分图最大匹配
```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int INF = 1e18;

int n, m, k;
vector<vector<int>> g;
vector<int> match;
vector<bool> vis;

bool dfs(int u) {
    for (int v : g[u]) {
        if (vis[v]) continue;
        vis[v] = true;
        if (match[v] == -1 || dfs(match[v])) {
            match[v] = u;
            return true;
        }
    }
    return false;
}

int hungary() {
    match.assign(m + 1, -1);
    int res = 0;
    for (int i = 1; i <= n; i++) {
        vis.assign(m + 1, false);
        if (dfs(i)) res++;
    }
    return res;
}

void solve() {
    cin >> n >> m >> k;
    g.resize(n + 1);
    
    for (int i = 1; i <= k; i++) {
        int u, v;
        cin >> u >> v;
        g[u].push_back(v);
    }
    
    cout << hungary() << '\n';
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--)
    solve();
    return 0;
}
```
