#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int INF = 1e18;

int n, m, q;
vector<int> w;
vector<vector<int>> edges;
vector<vector<int>> res;

void bfs(int sp) {
    queue<pair<int, int>> q;
    for (int i = 0; i < n; i++) {
        if (w[i] <= sp) {
            for (int j = sp; j <= 100; j++) res[j][i] = 0;
            q.push({i, 0});
        }
    }

    while (!q.empty()) {
        auto [i, j] = q.front();
        q.pop();
        if (j > res[sp][i]) continue;
        for (int k : edges[i]) {
            if (j + 1 >= res[sp][k]) continue;
            else {
                for (int f = sp; f <= 100; f++) res[f][k] = j + 1;
                q.push({k, res[sp][k]});
            }
        }
    }
}
 
void solve() {
    cin >> n >> m >> q;
    w.resize(n);
    edges.resize(n);
    res.resize(101, vector<int> (n, INF));
    for (int& i : w) cin >> i;
    while (m--) {
        int u, v; cin >> u >> v;
        u--, v--;
        edges[u].push_back(v);
        edges[v].push_back(u);
    }

    for (int i = 0; i <= 100; i++) {
        bfs(i);
    }

    while (q--) {
        int p, a; cin >> p >> a;
        p--;
        if (res[a][p] == INF) cout << -1 << '\n';
        else cout << res[a][p] << '\n';
    }
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    //int t; cin >> t;
    //while (t--) 
    solve();
}
