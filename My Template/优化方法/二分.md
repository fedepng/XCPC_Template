# 二分模板
```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int INF = 1e18;

int n;
vector<int> a;

// 二分查找：第一个 >= x 的位置（lower_bound）
int find_lower(int x) {
    int l = 1, r = n;
    while (l < r) {
        int mid = (l + r) / 2;
        if (a[mid] >= x) r = mid;
        else l = mid + 1;
    }
    return a[l] >= x ? l : n + 1;  // 没找到返回 n+1
}

// 二分查找：最后一个 <= x 的位置（upper_bound - 1）
int find_upper(int x) {
    int l = 1, r = n;
    while (l < r) {
        int mid = (l + r + 1) / 2;
        if (a[mid] <= x) l = mid;
        else r = mid - 1;
    }
    return a[l] <= x ? l : 0;  // 没找到返回 0
}

// 二分答案：找最小的满足 check(mid) >= x 的值
int find_answer_min(int x) {
    int l = 1, r = INF;
    int ans = -1;
    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(mid) >= x) {
            ans = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    return ans;
}

// 二分答案：找最大的满足 check(mid) <= x 的值
int find_answer_max(int x) {
    int l = 1, r = INF;
    int ans = -1;
    while (l <= r) {
        int mid = (l + r) / 2;
        if (check(mid) <= x) {
            ans = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }
    return ans;
}

// 实数二分
double find_real(double x) {
    double l = 0, r = INF;
    for (int i = 1; i <= 60; i++) {  // 循环 60 次保证精度
        double mid = (l + r) / 2;
        if (check(mid) >= x) r = mid;
        else l = mid;
    }
    return l;
}

void solve() {
    cin >> n;
    a.resize(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    sort(a.begin() + 1, a.end());
    
    int x;
    cin >> x;
    
    cout << find_lower(x) << '\n';
    cout << find_upper(x) << '\n';
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--)
    solve();
    return 0;
}
```
