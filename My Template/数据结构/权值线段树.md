# 权值线段树模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int INF = 1e18;
const int MAXV = 1e9;  // 值域范围

struct WeightedSegmentTree {
    vector<int> num, sum, l, r;
    int tot;
    
    WeightedSegmentTree(int size) {
        num.resize(size);
        sum.resize(size);
        l.resize(size);
        r.resize(size);
        tot = 0;
    }
    
    void update(int& rt, int segl, int segr, int pos, int val) {
        if (!rt) rt = ++tot;
        num[rt] += val;
        sum[rt] += val * pos;
        if (segl == segr) return;
        int mid = (segl + segr) >> 1;
        if (pos <= mid) update(l[rt], segl, mid, pos, val);
        else update(r[rt], mid + 1, segr, pos, val);
    }
    
    int kth(int rt, int segl, int segr, int k) {
        if (segl == segr) return segl;
        int mid = (segl + segr) >> 1;
        if (num[l[rt]] >= k) return kth(l[rt], segl, mid, k);
        else return kth(r[rt], mid + 1, segr, k - num[l[rt]]);
    }
    
    int query_cnt(int rt, int segl, int segr, int x) {
        if (!rt || segr <= x) return num[rt];
        int mid = (segl + segr) >> 1;
        if (x <= mid) return query_cnt(l[rt], segl, mid, x);
        else return num[l[rt]] + query_cnt(r[rt], mid + 1, segr, x);
    }
    
    int query_sum(int rt, int segl, int segr, int x) {
        if (!rt || segr <= x) return sum[rt];
        int mid = (segl + segr) >> 1;
        if (x <= mid) return query_sum(l[rt], segl, mid, x);
        else return sum[l[rt]] + query_sum(r[rt], mid + 1, segr, x);
    }
    
    int query_min_cnt(int rt, int segl, int segr, int x) {
        if (segl == segr) {
            return (x <= 0 ? 0 : (x + segl - 1) / segl);
        }
        int mid = (segl + segr) >> 1;
        if (sum[r[rt]] >= x) {
            return query_min_cnt(r[rt], mid + 1, segr, x);
        } else {
            return num[r[rt]] + query_min_cnt(l[rt], segl, mid, x - sum[r[rt]]);
        }
    }
};

void solve() {
    int n, q;
    cin >> n >> q;
    
    WeightedSegmentTree seg(200005 * 20);
    int root = 0;
    
    for (int i = 1; i <= n; i++) {
        int x;
        cin >> x;
        seg.update(root, 1, 1e9, x, 1);
    }
    
    while (q--) {
        int op, x;
        cin >> op >> x;
        if (op == 1) {
            seg.update(root, 1, 1e9, x, 1);
        } else if (op == 2) {
            seg.update(root, 1, 1e9, x, -1);
        } else if (op == 3) {
            cout << seg.kth(root, 1, 1e9, x) << '\n';
        }
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) 
    solve();
    return 0;
}
```
