# 快速幂模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int inf = 1e18;

int qpow(int a, int b) {
    int res = 1;
    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

void solve() {
    int a, b;
    cin >> a >> b;
    cout << qpow(a, b) << '\n';
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    //int t; cin >> t;

    solve();
    return 0;
}
```
