# 单调栈模板
可以以 $O(n)$ 复杂度求每个元素某侧第一个比它大/小的元素的下标或值。
```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int INF = 1e18;

int n;
vector<int> a;
vector<int> ne_l, ne_r;

void monotonic_stack() {
    stack<int> st;
    
    ne_r.assign(n, n);
    for (int i = 0; i < n; i++) {
        while (!st.empty() && a[i] >= a[st.top()]) {
            ne_r[st.top()] = i;
            st.pop();
        }
        st.push(i);
    }

    while (!st.empty()) st.pop();

    ne_l.assign(n, -1);
    for (int i = n - 1; i >= 0; i--) {
        while (!st.empty() && a[i] >= a[st.top()]) {
            ne_l[st.top()] = i;
            st.pop();
        }
        st.push(i);
    }
}

void solve() {
    cin >> n;
    a.resize(n);
    for (int& i : a) cin >> i;
    
    monotonic_stack();
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
