# 差分约束模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int INF = 1e18;

int n, m;
vector<int> dis, cnt;
vector<bool> vis;
vector<vector<pair<int, int>>> edges;

bool spfa() {
    queue<int> q;
    dis.assign(n + 1, INF);
    vis.assign(n + 1, 0);
    cnt.assign(n + 1, 0);
    
    dis[0] = 0;
    vis[0] = 1;
    q.push(0);
    
    while (!q.empty()) {
        int p = q.front();
        q.pop();
        vis[p] = 0;
        
        for (const auto& e : edges[p]) {
            int to = e.first;
            int w = e.second;
            if (dis[to] > dis[p] + w) {
                dis[to] = dis[p] + w;
                if (!vis[to]) {
                    vis[to] = 1;
                    q.push(to);
                    cnt[to]++;
                    if (cnt[to] > n) return 0;
                }
            }
        }
    }
    return 1;
}

void solve() {
    cin >> n >> m;
    edges.resize(n + 1);
    
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        // x_u - x_v ≤ w  =>  x_u ≤ x_v + w
        // 从 v 到 u 建边，权值为 w
        edges[v].push_back({u, w});
    }
    
    // 超级源点 0 连接到所有点，边权为 0
    for (int i = 1; i <= n; i++) {
        edges[0].push_back({i, 0});
    }
    
    if (spfa()) {
        for (int i = 1; i <= n; i++) {
            cout << dis[i] << " \n"[i == n];
        }
    } else {
        cout << "No\n";
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--)
    solve();
    return 0;
}
```
