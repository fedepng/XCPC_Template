# 线性基模板
```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int inf = 1e18;

struct LinearBasis {
    vector<int> basis;
    
    LinearBasis() {
        basis.resize(60, 0);
    }
    
    void insert(int x) {
        for (int i = 59; i >= 0; i--) {
            if (!(x >> i & 1)) continue;
            if (!basis[i]) {
                basis[i] = x;
                return;
            }
            x ^= basis[i];
        }
    }
    
    bool canXOR(int k) {
        for (int i = 59; i >= 0; i--) {
            if (!(k >> i & 1)) continue;
            if (!basis[i]) return 0;
            k ^= basis[i];
        }
        return 1;
    }
};
```
