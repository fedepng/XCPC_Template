#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;

int cpow(int a, int b) {
    int x = 1, y = a;
    while (b) {
        if (1 & b) x = (x * y) % mod;
        y = (y * y) % mod;
        b >>= 1;
    }
    return x;
}

void solve() {
    int n, k; cin >> n >> k;
    map<int, int> pw;
    set<int, greater<int>> s;
    for (int i = 0; i < n; i++) {
        int a, b; cin >> a >> b;
        pw[b] += a;
        s.insert(b);
    }
    for (auto it = pw.begin(); it != pw.end(); ) {
        int i = it->first;
        int& j = it->second;
        if (j > k) {
            int dis = j - k;
            j = k;
            pw[i + 1] += (dis + 1) / 2;
            if (dis % 2) j--;
            s.insert(i + 1);
        }
        if (j == 0) {
            it = pw.erase(it);
            s.erase(i);
        } else {
            ++it;
        }
    }
    map<int, int> ans;
    int a, b = 0, res = 0;
    for (int i : s) {
        if (pw[i] == 0) continue;
        if (!b) {
            a = k - pw[i];
            b = i;
            ans[i]++;
        } else {
            int dis = b - i;
            if (a && dis >= 45) break;
            while (dis) {
                a <<= 1; dis--;
                if (a > 2 * k) break;
            }
            if (a > 2 * k) break;
            if (a >= pw[i]) {
                a = a - pw[i];
                b = i;
            } else {
                ans[i]++;
                a = a + k - pw[i];
                b = i; 
            }
        }
    }
    for (auto [i, j] : ans) {
        res = (res + (j % mod) * cpow(2, i)) % mod;
    }
    cout << res << '\n';
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) 
    solve();
}
