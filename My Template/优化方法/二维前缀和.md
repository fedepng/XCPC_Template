# 二位前缀和模板

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int inf = 1e18;

int qsum(int x1, int y1, int x2, int y2, vector<vector<int>>& prefix) {
    return prefix[x2 + 1][y2 + 1] - prefix[x1][y2 + 1] - prefix[x2 + 1][y1] + prefix[x1][y1];
}

void solve() {
    int n; cin >> n;
    vector<vector<int>> a(n, vector<int>(n));
    vector<vector<int>> prefix(n + 1, vector<int>(n + 1, 0));
    for (auto& i : a) for (int& j : i) cin >> j;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            prefix[i + 1][j + 1] = prefix[i][j + 1] + prefix[i + 1][j] - prefix[i][j] + a[i][j];
        }
    }
    
    int x1 = 0, y1 = 0, x2 = 2, y2 = 2;
    int sum = qsum(x1, y1, x2, y2, prefix);
    cout << sum << '\n';
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    //int t; cin >> t;
    //while (t--) 
    solve();
}
```
