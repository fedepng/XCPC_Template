#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;

int n, m;
double p;
vector<int> a;

double cal(int k) {
    return (n * k + m) / (1 - pow(1.0 - p, k));
}

void solve() {
    cin >> n >> m >> p;
    p /= 10000.0;

    int h = 1;
    while (cal(h + 1) < cal(h)) h *= 2;

    int l = 1, r = h;
    double res = cal(r);
    while (l <= r) {
        int mid = l + (r - l) / 2;
        if (cal(mid + 1) > cal(mid)) {
            r = mid - 1;
            res = min(res, cal(mid));
        } else l = mid + 1;
    }

    cout << fixed << setprecision(8) << res << '\n';
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) 
    solve();
}
