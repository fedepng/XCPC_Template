# 单调队列模板
可以以 $O(n)$ 复杂度解决滑动窗口问题，求所有长度为k的子区间的最大元素。
```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int INF = 1e18;

int n, k;
vector<int> a;
deque<int> q;
vector<int> res;

void sliding_window_max() {
    q.clear();
    res.clear();
    
    for (int i = 1; i <= n; i++) {
        while (!q.empty() && q.front() < i - k + 1) {
            q.pop_front();
        }
        while (!q.empty() && a[q.back()] < a[i]) {
            q.pop_back();
        }
        q.push_back(i);
        if (i >= k) {
            res.push_back(a[q.front()]);
        }
    }
}

void solve() {
    cin >> n >> k;
    a.resize(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    sliding_window_max();
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--)
    solve();
    return 0;
}
```
