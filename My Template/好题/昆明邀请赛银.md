#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;

int gcd(int a, int b) {
    if (a == 0) return b;
    return gcd(b % a, a);
}

void solve() {
    int n, k; cin >> n >> k;
    vector<int> a(n);
    for (int& i : a) cin >> i;

    vector<int> pregcd(n + 1, 0), sufgcd(n + 1, 0);
    for (int i = 1; i <= n; i++) pregcd[i] = gcd(pregcd[i - 1], a[i - 1]);
    for (int i = n - 1; i >= 0; i--) sufgcd[i] = gcd(sufgcd[i + 1], a[i]);

    int res = pregcd[n];
    for (int i = 1; i <= n; i++) {
        if (pregcd[i] != pregcd[i - 1]) {
            int gc = pregcd[i - 1];
            for (int j = i; j <= n; j++) {
                gc = gcd(gc, a[j - 1] + k);
                res = max(res, gcd(gc, sufgcd[j]));
            }
        }
    }

    cout << res << '\n';
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) 
    solve();
}
