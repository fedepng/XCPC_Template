#include <bits/stdc++.h>
#define int long long
using namespace std;

const int mod = 998244353;
const int INF = 1e18;

int n;
vector<int> ti, a, sum, f;
vector<vector<int>> edges;

int dfs(int p, int fa) {
    int maxn0 = 0, p0 = -1, maxn1 = 0;
    for (int i : edges[p]) {
        if (i == fa) continue;
        if (ti[i] == 3) {
            if (a[i] > maxn0) {
                maxn1 = maxn0;
                maxn0 = a[i];
                p0 = i;
            } else if (a[i] > maxn1) {
                maxn1 = a[i];
            }
        } 
        sum[p] += dfs(i, p);
    }

    for (int i : edges[p]) {
        if (i == fa) continue;
        f[p] = max(f[p], sum[p] + a[i]); 
        if (i != p0) f[p] = max(f[p], sum[p] + a[i] - f[i] + sum[i] + maxn0);
        else f[p] = max(f[p], sum[p] + a[i] - f[i] + sum[i] + maxn1);
    }

    return f[p];
}
 
void solve() {
    cin >> n;
    edges.clear(), ti.clear(), a.clear(), sum.clear(), f.clear();
    ti.resize(n), a.resize(n), sum.resize(n, 0), f.resize(n, 0);
    edges.resize(n);
    for (int& i : a) cin >> i;
    for (int& i : ti) cin >> i;
    for (int i = 0; i < n - 1; i++) {
        int u, v; cin >> u >> v;
        u--, v--;
        edges[u].push_back(v);
        edges[v].push_back(u);
    }

    cout << dfs(0, -1) + a[0] << '\n';
} 
                   
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int t; cin >> t;
    while (t--) 
    solve();
}
